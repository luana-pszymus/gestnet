service:
  postgres_local:
    image: postgres:15-alpine
    container_name: gestnetdb
    enviroment:
      POSTGRES_USER: postgres
      POSTGRES_PASSWORD: postgres
      POSTGRES_DB: softwaredb
    ports:
      - "5433:5432"
    healthchek:
      test: ["CMD-SHELL", "pg_isready -U postgres -d softwaredb"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - gest-net

  backend:
    image: lupszymus/gestnet-backend:latest
    networks:
      - gest-net
    container_name: gestnet-back
    ports:
      - "3000:8000"
    depends_on:
      postgres_local:
        condition: service_healthy
    command: npm run dev
    restart: unless-stopped

  frontend:
    image: lupszymus/gestnet-frontend:latest
    container_name: gestnet-front
    ports:
      - "5173:5173"
    depends_on:
      - backend
    networks:
      - gest-net
    command: npm run dev
    restart: unless-stopped
